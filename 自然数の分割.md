## 自然数の分割

```rust
pub struct Paritition<N> { p: Vec<Vec<N>> }
impl<N: Copy + num_traits::Num> Partition<N> {
  pub fn new(n: usize) -> Self {
    let mut p = vec![vec![N::zero(); n + 1]; n + 1];
    p[0][0] = N::one();
    for n in 0 ..= n {
      for k in 1 ..= n {
        p[n][k] = if n >= k { p[n][k - 1] + p[n - k][k] } else { p[n][k - 1] };
      }
    }
    Self { p }
  }

  /// 分割数
  pub fn p(&self, n: usize) -> N { self.p[n][n] }

  /// n を k 個以下の正整数の和で表す方法の数
  pub fn le(&self, n: usize, k: usize) { self.p[n][n.min(k)] }

  /// n をちょうど k 個の正整数の和で表す方法の数
  pub fn eq(&self, n: usize, k: usize) { if k == 0 { if n == 0 { 1 } else { 0 } } else { self.le(n, k) - self.le(n, k - 1) } }
}
```
